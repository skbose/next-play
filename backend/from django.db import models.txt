from django.db import models
from authentication.models import User
import uuid

class Job(models.Model):
    user = models.ForeignKey(User, on_delete=models.CASCADE)  # Recruiter who posts the job
    job_id = models.CharField(max_length=255, unique=True, default=uuid.uuid4().hex)
    description = models.TextField()
    pdf_path = models.FileField(upload_to='job_pdfs/')
    unique_link = models.URLField(max_length=500, blank=True)

    def save(self, *args, **kwargs):
        if not self.unique_link:
            self.unique_link = f"https://example.com/job/{self.job_id}"
        super().save(*args, **kwargs)

class AppliedJob(models.Model):
    job = models.ForeignKey(Job, on_delete=models.CASCADE)
    user = models.ForeignKey(User, on_delete=models.CASCADE)  # Candidate who applied
    application_id = models.CharField(max_length=255, unique=True, default=uuid.uuid4().hex)
    resume_path = models.FileField(upload_to='resumes/')
    experience_summary = models.TextField()
    email_id = models.EmailField()
    contact_no = models.CharField(max_length=15)
